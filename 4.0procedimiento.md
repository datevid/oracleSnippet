
-- Declaracion
```
CREATE OR REPLACE PACKAGE MathOperations AS
    -- Declara una función pública que estará disponible fuera del paquete
    FUNCTION Square(n NUMBER) RETURN NUMBER;
END MathOperations;

--body
CREATE OR REPLACE PACKAGE BODY MathOperations AS

    -- Implementación de la función Square
    FUNCTION Square(n NUMBER) RETURN NUMBER IS
    BEGIN
        RETURN n * n;
    END Square;

END MathOperations;
```

--ejecucion
```
SELECT MathOperations.Square(3) FROM dual;
```

Creacion de un procedimiento con parametros de entrada que contengan array de datos
```
CREATE OR REPLACE PROCEDURE SASP_EJEMPLO_DOS_INSERT (
    PI_V_NOMBRE VARCHAR2(50),
    PI_N_EDAD NUMBER,
    PI_V_CIUDAD VARCHAR2(30),
    PI_V_CODIGO_POSTAL VARCHAR2(10),
    PI_A_VARCHAR2_ARRAY T_VARCHAR2_ARRAY -- Parámetro de entrada tipo array
) AS

    TYPE T_VARCHAR2_ARRAY IS TABLE OF VARCHAR2(50);

    V_N_ID_PERSONA NUMBER;

BEGIN

    -- Insertar en la tabla PERSONA
    INSERT INTO PERSONA (NOMBRE, EDAD, CIUDAD, CODIGO_POSTAL)
    VALUES (PI_V_NOMBRE, PI_N_EDAD, PI_V_CIUDAD, PI_V_CODIGO_POSTAL)
    RETURNING ID
    INTO V_N_ID_PERSONA;

    -- Insertar en la tabla TELEFONO
    FOR I IN 1 .. PI_A_VARCHAR2_ARRAY.COUNT LOOP

        INSERT INTO TELEFONO (ID_PERSONA, NUMERO_TELEFONO)
        VALUES (V_N_ID_PERSONA, PI_A_VARCHAR2_ARRAY(I));

    END LOOP;

    COMMIT;

EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        DBMS_OUTPUT.PUT_LINE('ERROR: ' || SQLERRM);
END SASP_EJEMPLO_DOS_INSERT;

DECLARE
    V_NOMBRE VARCHAR2(50) := 'Juan Pérez';
    V_EDAD NUMBER := 30;
    V_CIUDAD VARCHAR2(30) := 'Lima';
    V_CODIGO_POSTAL VARCHAR2(10) := '12345';
    V_VARCHAR2_ARRAY T_VARCHAR2_ARRAY := T_VARCHAR2_ARRAY('987654321', '123456789');
BEGIN

    SASP_EJEMPLO_DOS_INSERT(V_NOMBRE, V_EDAD, V_CIUDAD, V_CODIGO_POSTAL, V_VARCHAR2_ARRAY);

END;
```
